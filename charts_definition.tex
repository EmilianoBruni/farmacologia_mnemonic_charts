% Set for tkiz-qtree
\tikzset{
	drawarrow/.style={
		draw,->, thick,
		shorten <=0pt,
		shorten >=2pt
	},
	chartfont/.style={
		font=\ttfamily\normalsize\tiny,
		text centered,
	},
	chartnode/.style= {
		rectangle,
		chartfont,
		rounded corners,
		draw=black, very thick,
		minimum width=5em,
		minimum height=1.5em,
		align=center,
		inner sep=3pt,
	},
	grow'=right,level distance=100pt,sibling distance=6pt,
	every tree node/.style={
		chartnode
	},
	every level 0 node/.style={
		top color=white, bottom color=blue!30
	},
	edge from parent/.append style={
		drawarrow,edge from parent path={
			(\tikzparentnode) to[out=0,in=180] (\tikzchildnode)
		}
	},
		farmaco/.style={
		top color=white, bottom color=green!30
	},
	dummyc/.style = {
		circle,draw,top color=white, bottom color=white,
		inner sep=0pt, minimum size=7pt,
	},
	smallfont/.style={
		font=\tiny
	},
}

\def\upa{$\uparrow$}
\def\dwa{$\downarrow$}
